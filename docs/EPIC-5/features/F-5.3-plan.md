### Implementation Files
```typescript
// File: components/projects/ProjectSidebar.tsx
'use client';

import { useRouter } from 'next/navigation';
import { useState } from 'react';

interface NavItem {
  id: string;
  label: string;
  index: string;
  type: 'project' | 'redirect';
  target?: string;
}

export default function ProjectSidebar({ 
  selectedProject, 
  onProjectSelect 
}: {
  selectedProject: string | null;
  onProjectSelect: (projectId: string) => void;
}) {
  const router = useRouter();
  
  const PROJECTS_NAV: NavItem[] = [
    { id: 'betting-mgmt', label: 'BETTING_MGMT', index: '01', type: 'project' },
    { id: 'stakely', label: 'STAKELY', index: '02', type: 'project' },
    { id: 'school-bets', label: 'SCHOOL_BETS', index: '03', type: 'project' },
    { id: 'home', label: 'HOME', index: '04', type: 'redirect', target: '/' }
  ];

  const handleNavClick = (item: NavItem) => {
    if (item.type === 'redirect') {
      router.push(item.target!);
    } else {
      onProjectSelect(item.id);
    }
  };

  return (
    <div className="terminal-sidebar">
      <div className="sidebar-header">
        [ TERMINAL_NAV ]
        ------------------
      </div>
      
      {PROJECTS_NAV.map((item) => (
        <button
          key={item.id}
          onClick={() => handleNavClick(item)}
          className={`sidebar-nav-item ${
            selectedProject === item.id ? 'selected' : ''
          } ${item.type === 'redirect' ? 'redirect-item' : ''}`}
        >
          {item.type === 'redirect' ? (
            <span>[{item.index}] {item.label} (â†’ /)</span>
          ) : (
            <span>[{item.index}] {item.label}</span>
          )}
        </button>
      ))}
    </div>
  );
}
```
